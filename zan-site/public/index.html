game.zan
using Zan.Game;

class Pong : Game {
    Vec2 ball = new Vec2(480, 300);
    Vec2 vel  = new Vec2(4, 3);

    override void Update(float dt) {
        ball += vel;
        if (ball.Y < 0 || ball.Y > 600) vel.Y = -vel.Y;
    }

    override void Draw(Canvas c) {
        c.Clear(Color.Black);
        c.FillCircle(ball, 8, Color.White);
    }
}

# 克隆仓库Clone the repo
git clone https://github.com/joyspinch/zan-lang.git && cd zan-lang

# 构建编译器Build the compiler
mkdir build && cd build && cmake .. -DCMAKE_BUILD_TYPE=Release && cmake --build .

# 编译并运行第一个程序Compile & run your first program
./zanc hello.zan --auto-stdlib && ./hello
